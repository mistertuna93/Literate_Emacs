* Projectile
#+begin_src emacs-lisp
(use-package projectile
  :ensure t
  :diminish projectile-mode
  :init
  (projectile-mode +1)
  :config
  (setq projectile-completion-system 'default
        projectile-project-search-path '("~/projects/"))
  :bind-keymap
  ("C-c p" . projectile-command-map))
#+end_src
