#+begin_src emacs-lisp
(defun toggle-transparency ()
  "Toggle between transparent and opaque frame."
  (interactive)
  (let ((alpha (frame-parameter nil 'alpha)))
    (set-frame-parameter
     nil 'alpha
     (if (or (not alpha) (= (cdr alpha) 100))
         '(85 . 85) '(100 . 100)))))
#+end_src
